Manjaro 安装后的配置

本文最后更新于：2020年10月26日 上午

闲得无聊又把 Manjaro 装回来了。。正好这次做个笔记看都干了些啥。
仅适用于 Manjaro KDE，对于其他桌面环境仅供参考。

# Set Pacman Mirrors
sudo pacman-mirrors -i -c China -m rank
# Or do it yourself
sudo vim /etc/pacman.d/mirrorlist
# insert the mirror you like to the head of the file
# for example:
# Server = https://mirrors.ustc.edu.cn/archlinux/$repo/os/$arch

# Add archlinuxcn repo
sudo nano /etc/pacman.conf
## Add following into the back of the file
# [archlinuxcn]
# SigLevel = Optional TrustedOnly
# Server = https://mirrors.ustc.edu.cn/archlinuxcn/$arch

# Update repos && Upgrade all packages
sudo pacman -Syu archlinuxcn-keyring

# Upgrade kernel to latest linux-stable (which is 5.5 atm)
# linux55-nvidia-440xx is the driver for nvidia graphcis card (GTX 1070 here)
# headers and virtualbox-host-modules are optional
sudo pacman -S linux59 linux59-headers linux59-nvidia-440xx linux59-virtualbox-host-modules

# Remove old kernels (which is 5.4 atm)
sudo pacman -Rs linux58 linux58-nvidia-440xx

# Fuck default folder named in Chinese
nano .config/user-dirs.dirs
## Edit the following lines:
# XDG_DESKTOP_DIR="$HOME/Desktop"
# XDG_DOWNLOAD_DIR="$HOME/Download"
# XDG_TEMPLATES_DIR="$HOME/Templates"
# XDG_PUBLICSHARE_DIR="$HOME/Public"
# XDG_DOCUMENTS_DIR="$HOME/Documents"
# XDG_MUSIC_DIR="$HOME/Music"
# XDG_PICTURES_DIR="$HOME/Pictures"
# XDG_VIDEOS_DIR="$HOME/Videos"
## Then manually delete folders at $HOME and create
## corresponding folders named in English

# You should reboot now to apply folder name changes
reboot

# Dividing Line
echo "---------------------------------"

# Apply themes
## Install Adapta theme from official
## repositories
sudo pacman -S adapta-kde adapta-gtk-theme
## Then manually set themes in system
## settings. You can also downloads
## other themes directly in system
## settings

# Double click to open folders or files
# in Dolphin:
## 设置 - 工作空间行为 - 常规行为 - 点击行为 - 双击打开文件和文件夹

# zsh && oh-my-zsh &&
## As of Manjaro KDE 19.0.2, zsh is pre-installed,
## just install git and oh-my-zsh
sudo pacman -S git
sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"
## If forgot to set default term
## DO NOT USE sudo HERE SINCE IT WILL CHANGE THE DEFAULT SHELL OF ROOT!!!
chsh -s $(which zsh)
## I prefer this config: https://github.com/caiogondim/bullet-train.zsh
## Place it in $ZSH_CUSTOM/themes, here $ZSH_CUSTOM == ~/.zsh
mkdir -p ~/.zsh/themes
## Get bullet-train.zsh-theme in advance and locate to its position
mv bullet-train.zsh-theme ~/.zsh/themes
## Edit ~/.zshrc
nano ~/.zshrc
## Find ZSH_THEME and edit
# ZSH_THEME="bullet-train"
## Find ZSH_CUSTOM and remove comments
# ZSH_CUSTOM="$HOME/.zsh"
## Save the file.

# Use yay to install AUR packages
sudo pacman -S yay
# Optional: To build Lineage OS, install dependencies
yay -S lineageos-devel

# Optional: a lot of fonts incoming
## Get cascadia code from https://github.com/microsoft/cascadia-code/releases
## I prefer to download both versions, with or without powerline.
## And Sarasa Gothic from https://github.com/be5invis/Sarasa-Gothic/releases
## then install them manually
## Then noto-fonts and adobe-source-han
## (I know they are duplicated, but I'm fuckin' boring)
sudo pacman -S noto-fonts noto-fonts-cjk noto-fonts-emoji noto-fonts-extra
sudo pacman -S adobe-source-han-sans-otc-fonts adobe-source-han-serif-otc-fonts
sudo pacman -S adobe-source-code-pro-fonts
## Miscellaneous fonts
sudo pacman -S wqy-microhei wqy-microhei-lite wqy-zenhei ttf-roboto ttf-dejavu ttf-wps-fonts

# Fcitx && fcitx-sogoupinyin
sudo pacman -S fcitx-im kcm-fcitx fcitx-googlepinyin fcitx-skin-material

# Add profiles ~/.xprofile and reboot
echo "export GTK_IM_MODULES=fcitx\n\
export QT_IM_MODULES=fcitx\n\
export XMODIFIERS=\"@im=fcitx\"" >> ~/.xprofile

# Configure git
git config --global user.name ""
git config --global user.email ""
# Generate ssh key
ssh-keygen -t rsa -b 4096 -C "eterocell@outlook.com"
sudo pacman -S xclip
xclip =sel clip < ~/.ssh/id_rsa.pub
# Now ssh-key should have been copied to your paste board, set it in github
# Check connection
ssh -T git@github.com
# Generate a GPG key
gpg --full-gen-key
# Choose "RSA and RSA" "keysize = 4096" "0 = key does not expire"
# List generated key
gpg --list-secret-keys --keyid-format LONG
# `sec` stands for 'secret key', ssb stands for 'secret subkey'
# copy the public key and set in github
gpg --armor --export ${key ID which is after sec} | xclip --selection clipboard
# Then you can commit individual commits using
git commit --gpg-sign=${key ID which is after sec}
# Or just set all commits sign with it
git config --global user.signinkey ${key ID which is after sec}
git config --global commit.gpgsign true
# I've encrypted my SSH and GPG keys and stored online, convenient though risky

# Optional: Other packages
sudo pacman -S base-devel gcc gdb clang llvm vim google-chrome visual-studio-code-bin wps-office
sudo pacman -S nodejs npm maven gradle foxit-reader unrar unzip p7zip netease-cloud-music telegram-desktop
sudo pacman -S screenfetch neofetch
yay -S jetbrains-toolbox electron-ssr
# I recommend electron-ssr from https://github.com/shadowsocksrr/electron-ssr/releases
# Which is still mantained
